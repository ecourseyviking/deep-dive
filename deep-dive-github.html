function csvQ(val){

  const s=String(val===null||val===undefined?'':val);

  if(s.includes(',')||s.includes('"')||s.includes('\n')||s.includes('\r'))return'"'+s.replace(/"/g,'""')+'"';

function parseCSV(text){

  // Handles quoted fields, embedded commas, embedded newlines

  const rows=[];let i=0;const n=text.length;

  while(i<n){

    const row=[];

    while(i<n&&!(text[i]==='\r'||text[i]==='\n')||row.length===0&&i<n){

      if(i>=n||text[i]==='\r'||text[i]==='\n')break;

      if(text[i]==='"'){

        i++;let cell='';

        while(i<n){

          if(text[i]==='"'&&i+1<n&&text[i+1]==='"'){cell+='"';i+=2;}

          else if(text[i]==='"'){i++;break;}

          else{cell+=text[i++];}

        }

        row.push(cell);

        if(i<n&&text[i]===',')i++;

      } else {

        let cell='';

        while(i<n&&text[i]!==','&&text[i]!=='\r'&&text[i]!=='\n')cell+=text[i++];

        row.push(cell);

        if(i<n&&text[i]===',')i++;

      }

    }

    if(text[i]==='\r')i++;

    if(text[i]==='\n')i++;

    if(row.length>1||(row.length===1&&row[0]!==''))rows.push(row);

  }

  return rows;

}

  if(prog)prog.textContent='Importing...';

  const btn=document.querySelector('#modal-root .modal-dl-btn');

  if(btn)btn.disabled=true;


  const mode=(document.querySelector('input[name="imp-mode"]:checked')||{}).value||'merge';

  const {headers,matching}=window._imp;


  // Index all columns by header name

  const ci={};headers.forEach((h,i)=>{ci[h]=i;});


  // Parse dynamic EQ columns

  const eqCheckedCols=[],eqCommentCols=[];

  });

    for(let i=0;i<3;i++){s2focus.push(g(row,ci['S2.Focus '+(i+1)])||'');s2focusChecked.push((g(row,ci['S2.Focus '+(i+1)+' Checked'])||'').toLowerCase()==='yes');}

      createdBy:g(row,ci['Student'])||S.studentName,

      savedAt,collapsed:true,

      sourceType:g(row,ci['Source Type'])||'',

      sourceTitle:g(row,ci['Source Title'])||'',

      author:g(row,ci['Author/Org'])||'',

      url:g(row,ci['URL'])||'',

      mlaCitation:g(row,ci['MLA Citation'])||'',

      reliabilityNote:g(row,ci['Reliability Note'])||'',

      method:g(row,ci['Method'])||'',

      keyData:g(row,ci['Key Data/Results'])||'',

      studyConclusion:g(row,ci['Study Conclusion'])||'',

      mainArg:g(row,ci['Main Argument'])||'',

      evidence:g(row,ci['Supporting Evidence'])||'',

      advice:g(row,ci['Actionable Advice'])||'',

      eqs,eqComments,s2gen,s2genChecked,s2focus,s2focusChecked,quotes,

      otherNotes:g(row,ci['Other Notes'])||'',

      section:g(row,ci['Potential Section'])||'',

      tags

    };

  });


  if(mode==='replace'){

    S.cards=newCards;

  } else {

    // Merge: avoid duplicates by checking savedAt+title combo

    const existingKeys=new Set(S.cards.filter(c=>c.savedAt).map(c=>(c.savedAt||'')+'|'+(c.sourceTitle||'')));

    const deduped=newCards.filter(c=>!existingKeys.has((c.savedAt||'')+'|'+(c.sourceTitle||'')));

  }

  const n=newCards.length;

// Writes the current saved cards to localStorage so that if Drive

// loses data (e.g. the server trashed the old file but failed to

// write the new one), the student can recover on their next login

// without losing anything.


function backupToLocal(){

  if(!S.groupId)return;

  const saved=S.cards.filter(c=>c.savedAt);

  if(!saved.length)return;

  try{

    localStorage.setItem('dd-backup-'+S.groupId,JSON.stringify({

      cards:S.cards,

      backedUpAt:Date.now(),

      studentName:S.studentName

    }));

  }catch(e){/* localStorage unavailable — silent fail */}

}


function clearLocalBackup(gid){

  try{localStorage.removeItem('dd-backup-'+gid);}catch(e){}

}


function getLocalBackup(gid){

  try{const r=localStorage.getItem('dd-backup-'+gid);return r?JSON.parse(r):null;}

  catch(e){return null;}

}


async function checkAndRestoreBackup(gid){

  const backup=getLocalBackup(gid);

  if(!backup||!backup.cards||!backup.cards.length)return;


  const backupSaved=backup.cards.filter(c=>c.savedAt);

  if(!backupSaved.length){clearLocalBackup(gid);return;}


  const driveIds=new Set(S.cards.map(c=>c.id));

  const missingFromDrive=backupSaved.filter(c=>!driveIds.has(c.id));


  if(!missingFromDrive.length){

    // Backup is fully redundant — Drive has everything

    clearLocalBackup(gid);

    return;

  }


  if(S.cards.length===0){

    // Drive came back completely empty — auto-restore silently and save

    S.cards=backup.cards.filter(c=>c.savedAt);

    S.cards.forEach(padCard);

    S.driveRestoreNotice='auto';

    // Push everything back to Drive using bulk storageSet (safe write)

    try{

      await gs_('storageSet','dd-cards-'+gid,JSON.stringify(S.cards));

      clearLocalBackup(gid);

    }catch(e){

      // Drive still not writable — keep backup intact so next login can retry

    }

  } else {

    // Drive has SOME data but backup has extra cards not in Drive

    S.pendingRestore={cards:missingFromDrive,total:backupSaved.length,gid};

  }

}

// Collects all existing values from saved cards in the group.

function allGroupSections(){

  return [...new Set(S.cards.filter(c=>c.savedAt&&c.section).map(c=>c.section))].sort();

}

function allGroupTags(){

  return [...new Set(S.cards.filter(c=>c.savedAt).flatMap(c=>c.tags||[]))].sort();

}


function showSuggest(kind,id,val){

  const dropId=kind==='section'?'sec-suggest-'+id:'tag-suggest-'+id;

  const drop=document.getElementById(dropId);if(!drop)return;

  const raw=val.trim().replace(/^#/,'').toLowerCase();

  const pool=kind==='section'?allGroupSections():allGroupTags();


  // For tags, exclude ones already on this card

  let filtered=pool;

  if(kind==='tag'){

    const card=S.cards.find(c=>c.id===id);

    const current=card?card.tags:[];

    filtered=pool.filter(t=>!current.includes(t));

  }


  const matches=raw?filtered.filter(v=>v.toLowerCase().includes(raw)):filtered;


  if(!matches.length&&!raw){drop.classList.add('hidden');drop.innerHTML='';return;}


  let html='';

  matches.forEach((m,i)=>{

    const display=kind==='tag'?'#'+m:m;

    html+=`<div class="suggest-item" data-val="${esc(m)}" onmousedown="pickSuggest(event,'${kind}','${id}','${esc(m).replace(/'/g,"\'")}')">

      ${esc(display)}

    </div>`;

  });

  // "Add new" option when typed value doesn't exactly match anything

  const exactMatch=pool.some(v=>v.toLowerCase()===(raw));

  if(raw&&!exactMatch){

    const newLabel=kind==='tag'?'#'+val.trim().replace(/^#/,''):val.trim();

    html+=`<div class="suggest-item" data-val="${esc(val.trim().replace(/^#/,''))}" onmousedown="pickSuggest(event,'${kind}','${id}','${esc(val.trim().replace(/^#/,'')).replace(/'/g,"\'")}')">

      <span class="suggest-new">+ Add &ldquo;${esc(newLabel)}&rdquo;</span>

    </div>`;

  }


  if(!html){drop.classList.add('hidden');drop.innerHTML='';return;}

  drop.innerHTML=html;drop.classList.remove('hidden');

}


function hideSuggest(kind,id){

  const dropId=kind==='section'?'sec-suggest-'+id:'tag-suggest-'+id;

  const drop=document.getElementById(dropId);

  if(drop){drop.classList.add('hidden');drop.innerHTML='';}

}


function pickSuggest(e,kind,id,val){

  e.preventDefault();

  if(kind==='section'){

    const inp=document.getElementById(id+'-section');

    if(inp){inp.value=val;inp.dispatchEvent(new Event('input'));}

    hideSuggest('section',id);

  } else {

    commitTag(id,val);

    const inp=document.getElementById('taginput-'+id);

    if(inp){inp.value='';inp.focus();}

    hideSuggest('tag',id);

  }

}


function suggestKey(e,kind,id){

  const dropId=kind==='section'?'sec-suggest-'+id:'tag-suggest-'+id;

  const drop=document.getElementById(dropId);if(!drop||drop.classList.contains('hidden'))return;

  const items=[...drop.querySelectorAll('.suggest-item')];if(!items.length)return;

  const cur=drop.querySelector('.suggest-item.active');

  let idx=cur?items.indexOf(cur):-1;

  if(e.key==='ArrowDown'){e.preventDefault();idx=Math.min(idx+1,items.length-1);}

  else if(e.key==='ArrowUp'){e.preventDefault();idx=Math.max(idx-1,0);}

  else if(e.key==='Enter'&&cur){e.preventDefault();cur.dispatchEvent(new MouseEvent('mousedown'));return;}

  else if(e.key==='Escape'){hideSuggest(kind,id);return;}

  else return;

  items.forEach(el=>el.classList.remove('active'));

  if(idx>=0)items[idx].classList.add('active');

}
